------------------------------ Configure Log shipping

-- on FXLTCDB

SQL> ALTER SYSTEM SET LOG_ARCHIVE_DEST_2='SERVICE=MINERDB ASYNC NOREGISTER VALID_FOR=(ONLINE_LOGFILES,PRIMARY_ROLE) DB_UNIQUE_NAME=MINERDB';

SQL> ALTER SYSTEM SET LOG_ARCHIVE_DEST_STATE_2=ENABLE;

SQL> ALTER SYSTEM SET LOG_ARCHIVE_CONFIG='DG_CONFIG=(FXLTCDB,MINERDB)';

 

 

-- on MINERDB

SQL> ALTER SYSTEM SET log_archive_dest_1='LOCATION=/u02/oradata/MINERDB/archivelogs_local';

SQL> ALTER SYSTEM SET LOG_ARCHIVE_DEST_2='LOCATION=/u02/oradata/MINERDB/archivelogs VALID_FOR=(STANDBY_LOGFILE,PRIMARY_ROLE)';

SQL> ALTER SYSTEM SET LOG_ARCHIVE_DEST_STATE_2=ENABLE;

SQL> ALTER SYSTEM SET LOG_ARCHIVE_CONFIG='DG_CONFIG=(FXLTCDB,MINERDB)';
